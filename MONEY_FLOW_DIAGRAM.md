## SQL Query Visualization

```sql
-- This is what happens in the database

SELECT 
  -- Component 1: Raw Materials
  (SELECT SUM(total_cost) 
   FROM raw_material_purchases 
   WHERE store_id = 'your-store' 
     AND DATE(purchase_date) = CURRENT_DATE)  -- â‚¹1,500
  
  +  -- PLUS
  
  -- Component 2: Petty Cash
  (SELECT SUM(amount) 
   FROM petty_cash 
   WHERE store_id = 'your-store' 
     AND given_date = CURRENT_DATE)           -- â‚¹450
  
  +  -- PLUS
  
  -- Component 3: Expenses
  (SELECT SUM(amount) 
   FROM expenses 
   WHERE store_id = 'your-store' 
     AND expense_date = CURRENT_DATE)         -- â‚¹1,800

AS total_costs_today;                         -- = â‚¹3,750
```
